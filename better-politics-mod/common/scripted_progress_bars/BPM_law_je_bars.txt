je_bpm_multicultural_upgrade_progress_bar = {
	name = "je_bpm_multicultural_upgrade_progress_bar"
	desc = "je_bpm_multicultural_upgrade_progress_bar_desc"

	default_green = yes

    monthly_progress = {
		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_multicultural
					value < neutral
				}
				OR = {
					is_powerful = yes
					is_in_government = yes
				}
			}
			add = {
				desc = "bpm_multicultural_upgrade_from_anti_multiculturalists"
				add = -1
			}
		}
		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_multicultural
					value > neutral
				}
				OR = {
					is_powerful = yes
					is_in_government = yes
				}
			}
			add = {
				desc = "bpm_multicultural_upgrade_from_multiculturalists"
				add = 0.1
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 100
}
