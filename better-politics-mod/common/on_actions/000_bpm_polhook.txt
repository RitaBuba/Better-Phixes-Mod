on_monthly_pulse_country = {
    on_actions = {
        on_bpm_custom_elecend
    }
}

on_bpm_custom_elecend = {
    effect = {
        if = {
            limit = {
                has_variable = bpm_inelec
                in_election_campaign = no
            }
            remove_variable = bpm_inelec
            bpm_on_election_campaign_end_effect = yes
        }
        if = {
            limit = {
                NOT = { has_variable = bpm_inelec }
                in_election_campaign = yes
            }
            set_variable = bpm_inelec
        }
    }
}
