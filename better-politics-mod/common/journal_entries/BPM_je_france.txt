je_bpm_july_monarchy = {
	icon = "gfx/interface/icons/invention_icons/currency_standards.dds"

	group = je_group_historical_content

	immediate = {
		set_variable = {
			name = bpm_july_monarchy_prog_var
			value = 0
		}
	}

	is_progressing = {
		government_legitimacy >= 50
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.10
			}
		}
	}
	
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_civil_war
				trigger = {
					any_civil_war = {
						civil_war_progress > 0.10
					}
				}
			}
			triggered_desc = {
				desc = je_bpm_july_monarchy_paris_turmoil
				trigger = {
					capital = {
						turmoil >= 0.25
					}
				}
			}
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_fast
				trigger = {
					scope:journal_entry = {
						is_progressing = yes
					}
					government_legitimacy >= 75
				}
			}
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_normal
				trigger = {
					scope:journal_entry = {
						is_progressing = yes
					}
					government_legitimacy >= 50
				}
			}
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_low_legitimacy
				trigger = {
					government_legitimacy < 50
				}
			}
		}
	}
	
	complete = {
		custom_tooltip = {
			text = "bpm_je_progress_complete"
			has_variable = bpm_july_monarchy_prog_var
			var:bpm_july_monarchy_prog_var >= 100
		}
		OR = {
			has_law = law_type:law_census_voting
			has_law = law_type:law_universal_suffrage
		}
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.10
			}
		}
	}

	on_complete = {
		set_variable = {
			name = one_true_french_dynasty
		}
		trigger_event = { id = bpm_france.2 popup = yes }
	}

	fail = {
		OR = {
			NOT = { has_law = law_type:law_monarchy }
			capital = {
				turmoil >= 0.33
			}
			OR = {
				AND = {
					has_journal_entry = je_springtime_of_the_peoples
					radical_fraction = {
						value >= 0.25
					}
				}
				any_civil_war = {
					civil_war_progress > 0.75
				}
			}
		}
	}

	on_fail = {
		trigger_event = { id = bpm_france.1 popup = yes }
	}

	on_monthly_pulse = {
		events = {
			bpm_france_july_monarchy.1
		}
		effect = {
			if = {
				limit = {
					scope:journal_entry = {
						is_progressing = yes
					}
				}
				change_variable = {
					name = bpm_july_monarchy_prog_var
					add = {
						value = government_legitimacy
						multiply = 0.04
						subtract = 2
					}
				}
				clamp_variable = {
					name = bpm_july_monarchy_prog_var
					max = 100
					min = 0
				}
			}
		}
	}

	current_value = {
		value = var:bpm_july_monarchy_prog_var
	}

	goal_add_value = {
		value = 100
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}

je_bpm_nova_republica = {
	icon = "gfx/interface/icons/invention_icons/egalitarianism.dds"

	group = je_group_historical_content

	trigger = {
		
	}

	complete = {
		NOT = { has_law_or_variant = law_type:law_monarchy }
		NOT = {
			any_civil_war = {
				civil_war_progress >= 0.75
				is_civil_war_type = revolution
			}
		}
		NOT = {
			any_interest_group = {
				law_stance = {
					law = law_type:law_monarchy
					value > neutral
				}
				is_powerful = yes
			}
		}
		NOT = {
			any_interest_group = {
				leader = {
					has_ideology = ideology:ideology_bonapartist
				}
			}
			any_interest_group = {
				NOT = { is_interest_group_type = ig_reactionaries }
				leader = {
					has_ideology = ideology:ideology_legitimist
				}
			}
		}
		custom_tooltip = {
			text = "je_bpm_orleg_not_influencial_tooltip"
			bpm_orleg_clout =< 0.08
		}
	}

	on_complete = {
		trigger_event = { id = bpm_france.5 popup = yes }
	}

	fail = {
		NOT = {
			OR = {
				has_law = law_type:law_presidential_republic
				has_law = law_type:law_parliamentary_republic
				has_law = law_type:law_council_republic
			}
		}
	}

	should_be_pinned_by_default = yes
}
