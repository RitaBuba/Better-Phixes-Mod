	type = country
	random_valid = no
	
	text = {
		trigger = {
			OR = {
				has_global_variable = GLOBAL_VARIABLE_pope_pius_xi
				has_global_variable = GLOBAL_VARIABLE_pope_pius_x
				AND = {
					has_global_variable = GLOBAL_VARIABLE_pope_pius_ix

					has_global_variable = GLOBAL_VARIABLE_pope_pius_ix

		#fallback = yes

		trigger = {
			has_global_variable = GLOBAL_VARIABLE_pope_leo_xiii
		}

	}
